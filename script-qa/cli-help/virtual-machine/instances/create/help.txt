Create a Virtual Machine instance

Usage:
  ./cli virtual-machine instances create [flags]

Examples:
  ./cli virtual-machine instances create --image.id="xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx" --image.name="some_resource_name" --machine-type.id="xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx" --machine-type.name="some_resource_name" --network.associate-public-ip=true --network.vpc.id="xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx" --network.vpc.name="some_resource_name"

Flags:
      --availability-zone string              Availability Zone (min character count: 1)
      --cli.list-links enum[=table]           List all available links for this command (one of "json", "table" or "yaml")
  -h, --help                                  help for create
      --image object                          Image (at least one of: single property: id or single property: name)
                                              Use --image=help for more details
      --image.id string                       Image: Id (min character count: 1)
                                              This is the same as '--image=id:string'.
      --image.name string                     Image: Name (min character count: 1)
                                              This is the same as '--image=name:string'.
      --key-name string                       Key Name
      --machine-type object                   Machine Type (at least one of: single property: id or single property: name)
                                              Use --machine-type=help for more details
      --machine-type.id string                Machine Type: Id (min character count: 1)
                                              This is the same as '--machine-type=id:string'.
      --machine-type.name string              Machine Type: Name (min character count: 1)
                                              This is the same as '--machine-type=name:string'.
      --name string                           Name (between 1 and 225 characters)

                                              Use --network=help for more details (default {"associate_public_ip":true,"vpc":{"name":"default"}})

      --project-type enum                     An enumeration. (one of "dbaas", "default", "iamaas", "k8saas" or "mngsvc") (default "default")
